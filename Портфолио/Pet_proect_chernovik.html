def categorize(income):
    try:
        if 17 <= income <= 32:
            return ''
        elif 33 <= income <= 48:
            return ' '
        elif  income >= 49:
            return ''
    except:
        pass
data['category_age'] = data['age'].apply(categorize)

        if 0 <= income <= 40000:
            return 'C'
        elif 40001 <= income <= 80000:
            return 'B'
        elif income >= 80001:
            return 'A'
    except:
        pass
